320  // If the requested block is already mapped, we're done
321  if (block_index == current_mapped_block) {
322  return true;
323  }
324 
325  // Unmap previous block if any
326  if (mapped_ptr) {
327  ESP_LOGD(TAG, "- Unmapping block %u before mapping new block %u",
328  current_mapped_block, block_index);
329  unmap(); // Unmap previous block
330  }
331 
332  // Allocate map range
333  ESP_LOGD(TAG, "- Allocating map range for block %u", block_index);
334  esp_err_t err = esp_himem_alloc_map_range(ESP_HIMEM_BLKSZ, &range);

337  return false;
338  }
339 
340  // Map the current block
341  ESP_LOGD(TAG, "- Mapping block %u (offset %u)", block_index,
342  block_index * ESP_HIMEM_BLKSZ);
343  err = esp_himem_map(handle, range, block_index * ESP_HIMEM_BLKSZ, 0,
344  ESP_HIMEM_BLKSZ, ESP_HIMEM_PROT_RW, &mapped_ptr);
345  if (err != ESP_OK) {
346  ESP_LOGE(TAG, "Failed to map memory: %d", err);
347  esp_himem_free_map_range(range);
348  range = 0;
349  return false;
350  }
351 
352  current_mapped_block = block_index;
353  return true;

384  bool reallocate(size_t new_capacity) {
385  if (new_capacity <= element_capacity) {
386  return true; // No need to reallocate
387  }
388 
389  // Calculate new size (at least min_elements)
390  new_capacity = std::max(new_capacity, min_elements);
391 
392  // Create a new memory block
393  HimemBlock new_memory;
394  if (!new_memory.allocate(calculate_size_bytes(new_capacity))) {
395  ESP_LOGE(TAG, "Failed to allocate new memory block");
396  return false;
397  }
398 
399  // Copy existing elements if any
400  if (element_count > 0) {
401  T temp;

403  // Read from old memory
404  memory.read(&temp, i * sizeof(T), sizeof(T));
405

408  }
409  }

417  return true;
