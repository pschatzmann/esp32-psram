444  bool reallocate(size_t new_capacity) {
445  if (new_capacity <= element_capacity) {
446  return true; // No need to reallocate
447  }
448 
449  // Calculate new size (at least min_elements)
450  new_capacity = std::max(new_capacity, min_elements);
451 
452  // Create a new memory block
453  HimemBlock new_memory;
454  element_capacity = new_memory.allocate(calculate_size_bytes(new_capacity)) / sizeof(T);
455  if (element_capacity == 0) {
456  ESP_LOGE(TAG, "Failed to allocate new memory block");
457  return false;
458  }
459 
460  // Copy existing elements if any
461  if (element_count > 0) {
462  T temp;
463  for (size_t i = 0; i < element_count; ++i) {
464  // Read from old memory
465  memory.read(&temp, i * sizeof(T), sizeof(T));
466 
467  // Write to new memory

469  }
470  }
